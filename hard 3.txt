#include <stdio.h>

int countDigitOne(int n) {
    int count = 0;
    int factor = 1;
    int i = 1;

    while (i <= n) {
        int divisor = i * 10;
        count += (n / divisor) * i + ((n % divisor - i + 1) > 0 ? (n % divisor - i + 1) : 0);
        i *= 10;
    }

    return count;
}

int main() {
    int inputNumber = 13;
    int outputCount = countDigitOne(inputNumber);
    printf("%d\n", outputCount);

    return 0;
